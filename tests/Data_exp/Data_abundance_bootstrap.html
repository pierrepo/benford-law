def first_digit(data):
    """ Distribution des first digit des données
    
    Fonction permettant de renvoyer la distribution des first digit d'un jeu
    de données.
    
    Parameters
    ¯¯¯¯¯¯¯¯¯¯
    data : array of int
        Array d'entiers.
    
    Returns
    ¯¯¯¯¯¯¯
    fd : array
        Distribution des first digit.
    """
    fd = np.zeros(9, dtype=int)
    for x in range(len(data)):
        first = int(str(data[x])[0])
        fd[first-1] += 1
    return fd

def meanchi2(data_obs, freq_theo, nb_val, nb_loop, type_test):
    """ Moyenne des chi2 calculés.
    
    Fonction permettant de calculer la moyennes des chi2.
    
    parameters
    ¯¯¯¯¯¯¯¯¯¯
    data_obs : array of int
        Array d'entiers des données obsevés.
    freq_theo : array of float
        Array de float des fréquences théoriques.
    nb_val : int
        Nombre de d'individu par échantillon.
    nb_loop : int
        nombre de fois que l'on réalise la procédure de "bootstrap".
    
    Returns
    ¯¯¯¯¯¯¯
    float
        La moyenne des chi2 du "bootstrap".
    """
    sumChi2 = np.zeros(nb_loop, dtype=float)
    d_theo = np.array(freq_theo * nb_val)
    for i in range(nb_loop):
        ech = np.random.choice(data_obs, size=nb_val, replace=False)
        d_obs = first_digit(ech)
        sumChi2[i] = power_divergence(f_obs=d_obs, f_exp=d_theo, lambda_=type_test)[0]
    return (sum(sumChi2) / nb_loop)

def bootstrap_chi2(data_obs, freq_theo, nb_val, nb_loop, type_test):
    """ Moyenne des chi2 calculés et sa p-value associé."¶
    
    Fonction permettant de calculer la moyenne des chi2 et sa p-value associé.
    
    parameters
    ¯¯¯¯¯¯¯¯¯¯
    data_obs : array of int
        Array d'entiers des données obsevés.
    freq_theo : array of float
        Array de float des fréquences théoriques.
    nb_val : int
        Nombre de d'individu par échantillon.
    nb_loop : int
        nombre de fois que l'on réalise la procédure de "bootstrap".
    
    Returns
    ¯¯¯¯¯¯¯
    p_val : float
        p-value de mean_chi2
    mean_chi2 : float
        La moyenne des chi2 du "bootstrap". 
    """
    k = len(freq_theo+1)
    mean_chi2 = meanchi2(data_obs, freq_theo, nb_val, nb_loop, type_test)
    p_val = distributions.chi2.sf(mean_chi2, k - 1)
    return p_val, mean_chi2

conclusion¶

La procédure mise en place ("bootstrap") permet de montrer, por un rique alpha de 5%, que les données expérimentale d'abondances de protéines suivent ici la loi de Benford. Puisque la p-value obtenu est supérieur à 5%, on ne rejette pas l'hypothése $H_0$, donc la distribution théorique suit la loi Beford.

def size_ech(interval):
    """ création liste des tailles d'échantillons et/ou du nombre de loop.
    
    Fonction permettant de renvoyer une liste des tailles d'échantillons
    utilisé par la fonction evol_ech à partir d'une liste de 2 entiers
    contenant en 1ère position la taille minimum du 1er échantillon et en 2ème
    position la taille maximum du dernier échatillon.
    
    Parameters
    ¯¯¯¯¯¯¯¯¯¯
    interval : list of int
        liste de deux entiers.
        interval[0] : position taille minimum du 1er échantillon
        interval[1] : position taille maximum du dernier échantillon
        
    Returns
    ¯¯¯¯¯¯¯
    list_val : list of int
        liste contenant la taille des echantillons pour la procédure
        "bootstrap".
    """
    if len(interval) != 2:
        return print("Erreur, mauvaise définition de l_val : taille max de l_val = 2\n")
    elif interval[0] > interval[1]:
        return print("Erreur, mauvaise définition de l_val : l_val[0] = min > l_val[1] = max\n")
    elif interval[0] == interval[1]:
        return print("Erreur, mauvaise définition de l_val : l_val[0]=min == l_val[1]=max\n")
    list_val = []
    if interval[0] < 1_000 and interval[1] <= 1_000:
        list_val += list(range(interval[0], interval[1]+100, 100))
    elif interval[0] >= 1_000 and interval[1] > 1_000:
        list_val += list(range(interval[0], interval[1]+1_000, 1_000))
    elif interval[0] < 1_000 and interval[1] > 1_000:
        list_val += list(range(interval[0], 1_000, 100))
        list_val += list(range(1_000, interval[1]+1_000, 1_000))
    else:
        return print("Erreur\n")
    return list_val

def evol_ech(data_obs, freq_theo, l_val, l_loop):
    """ Évolution du chi2 calculé et de la p-value
    
    Fonction permettant de suivre l'évolution du chi2 calculé et de la p-value
    en fonction de la taille de l'échantillon et/ou du nombre de "bootstrap".
    
    Parameters
    ¯¯¯¯¯¯¯¯¯¯
    data_obs : array of int
        Array d'entiers des données obsevés.
    freq_theo : array of float
         Array de float des fréquences théoriques.
    l_val : list of int or int
        Liste de 2 entiers contenant en 1ère position la taille minimum de
        l'échantillon et en 2ème position la taille maximum de l'échantillon,
        ou taille des échantillons du "bootstrap".
    l_loop : list of int or int
        Liste de 2 entiers contenant en 1ère position le nombre minimum de
        "bootstrap" et en 2ème position le nombre maximum de "bootstrap", ou
        nombre de loop de la procédure "bootstrap".
        
    Returns
    ¯¯¯¯¯¯¯
    DataFrame
    """
    if type(l_val) == type(l_loop) == int:
        return print("Erreur, utilisé la fonction : bootstrap_chi2(data_obs, freq_theo, nb_val, nb_loop)\n")
    
    chi2 = []
    pval = []
    
    elif type(l_val) == type(l_loop) == list:
        list_val = []
        list_loop = []
        l_val = size_ech(l_val)
        l_loop = size_ech(l_loop)
        for i in range(len(l_loop)):
            for j in range(len(l_val)):
                chi2_pval = bootstrap_chi2(exp_numbers, freq_benford, l_val[j], l_loop[i])
                chi2.append(chi2_pval[1])
                pval.append(chi2_pval[0])
                list_val.append(l_val[j])
                list_loop.append(l_loop[i])
        return pd.DataFrame(data={"Numbers of loop":list_loop,"Sample size":list_val,"Chi2":chi2, "p-value":pval})
                    
    elif type(l_val) == int and type(l_loop) == list:
        l_loop = size_ech(l_loop)
        for i in range(len(l_loop)):
            chi2_pval = bootstrap_chi2(data_obs, freq_theo, l_val, l_loop[i])
            chi2.append(chi2_pval[1])
            pval.append(chi2_pval[0])
        return pd.DataFrame(data={"Numbers of loop":l_loop, "Chi2":chi2, "p-value":pval})
        
    elif type(l_val) == list and type(l_loop) == int:
        l_val = size_ech(l_val)
        for i in range(len(list_val)):
            chi2_pval = bootstrap_chi2(data_obs, freq_theo, l_val[i], l_loop)
            chi2.append(chi2_pval[1])
            pval.append(chi2_pval[0])
        return pd.DataFrame(data={"Sample size":l_val, "Chi2":chi2, "p-value":pval})
        
    else:
        return print("Erreur, \n")
